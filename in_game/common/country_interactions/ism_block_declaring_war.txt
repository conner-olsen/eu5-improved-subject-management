block_subject_declaring_war = {
	type = subject
	category = CATEGORY_SUBJECT_ACTIONS

	potential = {
	}

	allow = {
	}

	ai_tick = never

	use_enroute = no

	select_trigger = {
		looking_for_a = country
		interaction_source_list = {
			scope:actor = {
				every_subject = {
					add_to_list = source
				}
			}
		}
		target_flag = recipient
		name = "choose_country"
		column = {
			data = name
		}
		visible = {
			NOT = { has_country_modifier = overlord_blocking_declaring_war }
		}
        enabled = {
		}
	}

	effect = {
		if = {
			limit = {
				exists = scope:recipient
			}
			scope:recipient = {
				add_country_modifier = {
					modifier = overlord_blocking_declaring_war
					days = -1
					mode = replace
				}
			}
		}
	}
}
