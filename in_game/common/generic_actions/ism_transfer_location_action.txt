    type = owncountry
    show_message = no

    automation_tick = never

    potential = {  }
    allow = {  }

    select_trigger = {
        looking_for_a = location
        interaction_source_list = {
            scope:actor.var:ism_giver_nation = {
                every_owned_location = {
                    add_to_list = source
                }
            }
        }
        target_flag = recipient
        source_flags = only_actual_locations
        name = "transfer_location_to_subject_select_title"
        none_available_msg_key = "no_valid_locations"

        column = { data = name }

        visible = {
            can_transfer_location = {
                giver = scope:actor.var:ism_giver_nation
                receiver = scope:actor.var:ism_receiver_nation
            }
        }

        enabled = {
            custom_tooltip = {
                text = cannot_transfer_capital
                ism_allow_transfer_location = yes
            }
        }
    }

    effect = {
        if = {
            limit = {
                exists = scope:recipient

            ism_transfer_location_effect = {
                giver = scope:actor.var:ism_giver_nation
                receiver = scope:actor.var:ism_receiver_nation
                target_location = scope:recipient
            }
            hidden_effect = {
                fire_generic_action = {
                    type = ism_transfer_location_action
                    actor = scope:actor
                }
            }
        }
