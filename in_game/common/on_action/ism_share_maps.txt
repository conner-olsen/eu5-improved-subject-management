yearly_country_pulse = {
    on_actions = {
        ism_share_maps
    }
}

ism_share_maps = {
    trigger = {
        is_overlord = yes
    }
    effect = {
        save_temporary_scope_as = top_overlord
        every_area = {
            limit = {
                NOT = { scope:top_overlord = { has_discovered_area = prev } }
                scope:top_overlord = { any_subject_or_below = { has_discovered_area = prev.prev } }
            }
            scope:top_overlord = { discover_area = prev }
        }
    }
}
